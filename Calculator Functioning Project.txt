#include <iostream>
#include <cmath> 
#include <string>
using namespace std;

int main () {
    string operation;
    double num1, num2;
    cout << "Enter a number for calculation: " ;
    cin >> num1;
    cout << "Enter the operation for calculation ( +, -, *, /, ^ ): " ;
    cin >> operation;
    cout << "Enter your second number for calculation: " ;
    cin >> num2;
    

    // operations for your calculations
    if (operation == "+") {
        cout << num1 + num2; }
        else if (operation == "-") {
            cout << num1 - num2; }
            else if (operation == "*") {
                cout << num1 * num2; }
                else if (operation == "/") {
                    if (num1 == 0) {
                        cout << "Error, cannot divide by zero. Please try again."; 
                    }
                    else { cout << num1 / num2; }
                }
                    else if (operation == "^") {
                        cout << pow(num1, num2) ; }
                                        else {
                        cout << "Invalid operation entered. Please try again.";

                    }
                    return 0; }
